Test1 := function(N,Neg,M,H,L,K);
if L!N eq -L!K or L!N eq -2*L!K then return [];
end if;
S := [Z : Z in DisjointCurves(Neg,M,L,K) | qua(L!N,L!Z[1],M) eq 0];
X :=[];
for Z in S do
N2:=L!N-L!Z[2];
if IsEff(N2,Neg,M,H) then
if Nefification(L!Z[1],N2,Neg,H,M,L)
then X := X cat [[L!N,L!Z[1],L!Z[2]]];
end if;
end if;
end for;
return X;
end function;
